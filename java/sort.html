        CompareByPrice.java

        import java.util.Comparator;

        public class CompareByPrice implements Comparator {

        public int compare(infoBean o1, infoBean o2) {

            int price1 = Integer.parseInt(o1.getPrice());
            int price2 = Integer.parseInt(o2.getPrice());

            if( price1 > price2 ){

                //昇順のソート
                //return 1;
                //降順のソート
                return -1;

            }else if( price1 < price2 ){

                //昇順ソートの場合
                //return -1;
                //降順ソートの場合
                return 1;

            }

            //等価の場合
            return 0;


            }

        }
