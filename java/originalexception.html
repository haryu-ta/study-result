            public class originalException extends Exception {

                private int err_code;
                private Sting err_cause;

                public originalException ( int err_code,String message,String err_cause ) {
                    super(message);
                    this.err_code = err_code;
                    this.err_cause = err_cause;
                }

                public int getErr_code(){
                    return err_code;
                }

                public String getErr_cause(){
                    return err_cause;
                }

            }
