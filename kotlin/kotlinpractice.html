            ＠Autowired
            lateinit private val resourceLoader : ResourceLoader

            val resource: Resource = resourceLoader.getResource("file:[ファイル名]")
            val bf : BufferedReader = BufferedReader(FileReader(resource.file))
            bf.use{
                line = it.readLine()
            }
            // useを使うとclose不要
