Stateの非同期更新

            
            import { useState } from "react";

            const Example = () => {
                // useStateは関数の一番上でしか使えない
                const [ count,setCount ] = useState(0);
            
                const clickBtn = () => {
                    // console.log(`前状態: ${count}`); 
                    // setCount(count + 1);
                    // //非同期的に処理されるため、初期状態
                    // console.log(`後状態: ${count}`);
                    // 前状態と後状態は同じ値になる
                    setCount(count + 1);
                    setCount( prevstate => {
                        return prevstate + 1;
                    });        
                }
            
                return (
                    <>
                        <p>カウント数: {count}</p>
                        <button
                            onClick={clickBtn}>
                        +
                        </button>
                    </>
                );
            };
            
            export default Example;
