import java.io.IOException;
            import java.util.Enumeration;
            import java.util.HashMap;
            import java.util.ResourceBundle;

            public class Kanji {

                private static String confpath = "kanji2";

                public static String convert(String kanji) throws IOException {

                     HashMap name = readConfig();
                     String returnkanji = "";

                     if( name.containsKey(kanji) ){

                        returnkanji = name.get(kanji);

                     }else{

                         returnkanji = kanji;

                     }

                    return returnkanji;

                }

                private static HashMap readConfig() throws IOException {

                    HashMap map = new HashMap();

                    ResourceBundle rb = ResourceBundle.getBundle(confpath);
                    Enumeration em = rb.getKeys();
                    String key;
                    while ( em.hasMoreElements() ){
                        key=em.nextElement();
                        map.put(key,rb.getString(key));
                    }


                    return map;
                }

            }
